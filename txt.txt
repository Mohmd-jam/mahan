        // Counter Animation
        function animateCounters() {
            const counters = [
                { element: document.getElementById('projects-count'), target: 42, duration: 2000 },
                { element: document.getElementById('experience-years'), target: 15, duration: 2000 },
                { element: document.getElementById('clients-count'), target: 120, duration: 2000 },
                { element: document.getElementById('awards-count'), target: 8, duration: 2000 }
            ];

            counters.forEach(counter => {
                let current = 0;
                const increment = counter.target / (counter.duration / 16);

                const updateCounter = () => {
                    current += increment;
                    if (current < counter.target) {
                        counter.element.textContent = Math.floor(current);
                        requestAnimationFrame(updateCounter);
                    } else {
                        counter.element.textContent = counter.target;
                    }
                };

                updateCounter();
            });
        }

        // Modal Functions
        function openModal(modalId) {
            document.getElementById(modalId).classList.add('active');
            document.body.style.overflow = 'hidden';
        }

        function closeModal(modalId) {
            document.getElementById(modalId).classList.remove('active');
            document.body.style.overflow = 'auto';
        }

        // Close modal when clicking outside
        window.onclick = function (event) {
            if (event.target.classList.contains('modal')) {
                event.target.classList.remove('active');
                document.body.style.overflow = 'auto';
            }
        }

        // Contact Form Logic
        document.addEventListener('DOMContentLoaded', function () {
            // Animate counters
            animateCounters();

            // Contact form submit
            const contactForm = document.querySelector('form.bg-white');
            if (contactForm) {
                contactForm.addEventListener('submit', function (e) {
                    e.preventDefault();
                    // Get values
                    const name = document.getElementById('name').value.trim();
                    const email = document.getElementById('email').value.trim();
                    const phone = document.getElementById('phone').value.trim();
                    const message = document.getElementById('message').value.trim();
                    // Simple validation
                    if (!name || !email || !phone || !message) {
                        showToast('لطفاً همه فیلدها را پر کنید.', false);
                        return;
                    }
                    if (!validateEmail(email)) {
                        showToast('ایمیل وارد شده معتبر نیست.', false);
                        return;
                    }
                    // Success (simulate send)
                    showToast('پیام شما با موفقیت ارسال شد!', true);
                    contactForm.reset();
                });
            }

            // Newsletter submit
            const newsletterInput = document.querySelector('footer input[type="email"]');
            const newsletterBtn = document.querySelector('footer button');
            if (newsletterInput && newsletterBtn) {
                newsletterBtn.addEventListener('click', function (e) {
                    e.preventDefault();
                    const email = newsletterInput.value.trim();
                    if (!email) {
                        showToast('لطفاً ایمیل خود را وارد کنید.', false);
                        return;
                    }
                    if (!validateEmail(email)) {
                        showToast('ایمیل وارد شده معتبر نیست.', false);
                        return;
                    }
                    showToast('ایمیل شما با موفقیت ثبت شد!', true);
                    newsletterInput.value = '';
                });
            }

            // Scroll to projects section
            const viewAllBtn = document.querySelector('button.gold-bg.text-white.rounded-full.font-semibold');
            const projectsSection = document.getElementById('projects');
            if (viewAllBtn && projectsSection) {
                viewAllBtn.addEventListener('click', function () {
                    projectsSection.scrollIntoView({ behavior: 'smooth' });
                });
            }
        });

        // Email validation helper
        function validateEmail(email) {
            return /^\S+@\S+\.\S+$/.test(email);
        }

        // Toast message
        function showToast(message, success) {
            let toast = document.getElementById('toast-message');
            if (!toast) {
                toast = document.createElement('div');
                toast.id = 'toast-message';
                toast.style.position = 'fixed';
                toast.style.bottom = '40px';
                toast.style.left = '50%';
                toast.style.transform = 'translateX(-50%)';
                toast.style.zIndex = '9999';
                toast.style.minWidth = '220px';
                toast.style.textAlign = 'center';
                toast.style.padding = '16px 24px';
                toast.style.borderRadius = '12px';
                toast.style.fontSize = '1rem';
                toast.style.boxShadow = '0 4px 16px rgba(0,0,0,0.15)';
                document.body.appendChild(toast);
            }
            toast.textContent = message;
            toast.style.background = success ? '#D4AF37' : '#e53e3e';
            toast.style.color = success ? '#fff' : '#fff';
            toast.style.opacity = '1';
            toast.style.pointerEvents = 'auto';
            setTimeout(() => {
                toast.style.opacity = '0';
                toast.style.pointerEvents = 'none';
            }, 3000);
        }
